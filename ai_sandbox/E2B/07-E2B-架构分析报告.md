# E2B (Code-Interpreter) 架构分析报告

## 项目概述

E2B (Execute to Build) 是专为 AI 应用设计的云代码执行沙箱平台,通过 Firecracker 微虚拟机提供安全、快速、可扩展的代码运行环境。它是 AI Agent 领域的关键基础设施,让 LLM 能够安全地执行用户生成的代码、访问文件系统、运行长时间任务。

**项目定位:**
- **场景**: AI Code Interpreter, LLM Tool Use, 代码执行沙箱
- **隔离方式**: Firecracker microVM (硬件虚拟化)
- **设计理念**: 安全优先 + 开发者友好 + 性能优化

**核心特点:**
- ✅ **多语言支持**: Python, JavaScript, TypeScript, Bash 等
- ✅ **流式执行**: 实时输出 stdout/stderr,支持长时间运行
- ✅ **文件系统操作**: 完整的文件 CRUD + 目录监控
- ✅ **网络控制**: 细粒度出站白名单/黑名单
- ✅ **模板系统**: Docker-based 自定义环境
- ✅ **毫秒级启动**: 通过 VM 快照优化冷启动

**技术栈:**
- **虚拟化**: Firecracker microVM
- **协议**: gRPC Connect (HTTP/2) + REST API
- **SDK**: Python (v2.10.2), JavaScript/TypeScript (v2.10.3)
- **语言**: Python 3.11.6, Node.js 20.9.0

### 1.2 核心特点解析

#### 特点一: 流式代码执行 - 实时反馈与长任务支持

**设计目标:** LLM 需要实时看到代码执行输出,而非等待完成后一次性返回

**流式协议设计** (`spec/proto/process.proto`):

```protobuf
service Process {
  // 启动进程并流式返回事件
  rpc Start(ProcessStartRequest) returns (stream ProcessEvent);

  // 连接到运行中的进程
  rpc Connect(ProcessConnectRequest) returns (stream ProcessEvent);

  // 发送输入到进程
  rpc SendInput(ProcessSendInputRequest) returns (ProcessSendInputResponse);
  rpc StreamInput(stream ProcessStreamInputRequest) returns (ProcessStreamInputResponse);

  // 发送信号
  rpc SendSignal(ProcessSendSignalRequest) returns (ProcessSendSignalResponse);

  // 列出所有进程
  rpc List(ProcessListRequest) returns (ProcessListResponse);
}

// 进程事件流
message ProcessEvent {
  oneof event {
    StartEvent start;       // 进程已启动
    DataEvent data;         // 输出数据
    EndEvent end;           // 进程已结束
    KeepAlive keepalive;    // 保活心跳
  }
}

message StartEvent {
  string pid = 1;  // 进程 ID
}

message DataEvent {
  oneof output {
    bytes stdout = 1;  // 标准输出
    bytes stderr = 2;  // 标准错误
    bytes pty = 3;     // 伪终端输出 (PTY)
  }
}

message EndEvent {
  sint32 exit_code = 1;   // 退出码
  bool exited = 2;        // 是否正常退出
  string status = 3;      // 状态描述
  optional string error = 4;  // 错误信息
}

message KeepAlive {}  // 每 50 秒发送一次
```

**Python SDK 流式接口** (`packages/python-sdk/e2b/sandbox/commands.py`):

```python
class CommandHandle:
    """命令执行句柄,支持流式读取输出"""

    def __init__(self, process_stream):
        self._stream = process_stream
        self._stdout_buffer = []
        self._stderr_buffer = []
        self._exit_code = None
        self._done = False

    async def _iterate_events(self) -> AsyncIterator[ProcessEvent]:
        """迭代进程事件流"""
        async for event in self._stream:
            if event.start:
                # 进程已启动
                self.pid = event.start.pid

            elif event.data:
                # 数据事件
                if event.data.stdout:
                    self._stdout_buffer.append(event.data.stdout.decode())
                    yield ("stdout", event.data.stdout.decode(), None)

                elif event.data.stderr:
                    self._stderr_buffer.append(event.data.stderr.decode())
                    yield ("stderr", None, event.data.stderr.decode())

                elif event.data.pty:
                    yield ("pty", None, None, event.data.pty.decode())

            elif event.end:
                # 进程结束
                self._exit_code = event.end.exit_code
                self._done = True
                if event.end.error:
                    raise RuntimeError(event.end.error)
                break

            elif event.keepalive:
                # 保活心跳,忽略
                continue

    async def wait(self, timeout: Optional[float] = None) -> ExecutionResult:
        """等待进程完成,返回完整结果"""
        start_time = time.time()

        async for event_type, stdout, stderr in self._iterate_events():
            if timeout and time.time() - start_time > timeout:
                await self.kill()
                raise TimeoutError(f"Command timed out after {timeout}s")

        return ExecutionResult(
            stdout="".join(self._stdout_buffer),
            stderr="".join(self._stderr_buffer),
            exit_code=self._exit_code,
        )

    @property
    def stdout(self) -> str:
        """累积的标准输出"""
        return "".join(self._stdout_buffer)

    @property
    def stderr(self) -> str:
        """累积的标准错误"""
        return "".join(self._stderr_buffer)

    @property
    def exit_code(self) -> Optional[int]:
        """退出码 (进程未结束时为 None)"""
        return self._exit_code

    async def kill(self):
        """强制终止进程 (SIGKILL)"""
        await self._sandbox.commands.send_signal(self.pid, Signal.SIGKILL)

    async def terminate(self):
        """优雅终止进程 (SIGTERM)"""
        await self._sandbox.commands.send_signal(self.pid, Signal.SIGTERM)


class Commands:
    """命令执行管理器"""

    async def run(self,
                  cmd: str,
                  on_stdout: Optional[Callable[[str], None]] = None,
                  on_stderr: Optional[Callable[[str], None]] = None,
                  background: bool = False,
                  cwd: Optional[str] = None,
                  envs: Optional[Dict[str, str]] = None,
                  timeout: Optional[float] = 60.0) -> CommandHandle:
        """
        运行命令

        Args:
            cmd: 命令字符串
            on_stdout: 标准输出回调 (实时调用)
            on_stderr: 标准错误回调 (实时调用)
            background: 是否后台运行 (不等待完成)
            cwd: 工作目录
            envs: 环境变量
            timeout: 超时时间 (秒)

        Returns:
            CommandHandle 对象
        """
        # 构造 gRPC 请求
        request = ProcessStartRequest(
            cmd=cmd,
            cwd=cwd or "/",
            envs=envs or {},
            timeout_ms=int(timeout * 1000) if timeout else 60000,
        )

        # 发起流式调用
        stream = self._grpc_client.Start(request)
        handle = CommandHandle(stream)

        # 如果提供了回调,启动后台任务处理流
        if on_stdout or on_stderr:
            async def process_stream():
                async for event_type, stdout, stderr in handle._iterate_events():
                    if stdout and on_stdout:
                        on_stdout(stdout)
                    if stderr and on_stderr:
                        on_stderr(stderr)

            asyncio.create_task(process_stream())

        # 后台模式: 立即返回,不等待完成
        if background:
            return handle

        # 前台模式: 等待完成
        await handle.wait(timeout=timeout)
        return handle
```

**使用示例:**

```python

from e2b import AsyncSandbox

async def main():
    sandbox = await AsyncSandbox.create(template="base")

    # 方式 1: 回调式流式输出
    def on_stdout(output: str):
        print(f"[OUT] {output}", end="")

    def on_stderr(output: str):
        print(f"[ERR] {output}", end="", file=sys.stderr)

    await sandbox.commands.run(
        "pip install numpy && python -c 'import numpy; print(numpy.__version__)'",
        on_stdout=on_stdout,
        on_stderr=on_stderr,
        timeout=120

    # 方式 2: 迭代器式流式输出
    handle = await sandbox.commands.start("find / -name '*.py' | head -10")
    async for event_type, stdout, stderr in handle._iterate_events():
        if stdout:
            print(stdout, end="")

    # 方式 3: 等待完成后一次性获取
    result = await sandbox.commands.run("ls -la /tmp")
    print(result.stdout)
    print(f"Exit code: {result.exit_code}")

    await sandbox.kill()

asyncio.run(main())
```

**Keep-Alive 机制:**

E2B 使用 50 秒的 keep-alive 心跳防止长时间运行的进程被误判为超时:


**优势:**

- **长任务支持**: 支持数小时的长时间执行(如模型训练、数据处理)
- **可中断性**: 随时发送 SIGTERM/SIGKILL 终止失控的进程
- **多进程管理**: 支持并发运行多个进程,通过 PID 区分

**Python SDK 文件操作** (`packages/python-sdk/e2b/sandbox/filesystem.py`):

```python
class Filesystem:
    """文件系统操作管理器"""

    async def read(self,
                   path: str,
                   format: Literal["text", "bytes"] = "text",
                   user: Optional[str] = None) -> Union[str, bytes]:
        """
        读取文件内容

        Args:
            path: 文件路径
            format: 返回格式 ("text" 或 "bytes")
            user: 运行用户 (默认当前用户)

        Returns:
            文件内容
        """
        # 使用 cat 命令读取
        cmd = f"cat {path}"
        if user:
            cmd = f"sudo -u {user} {cmd}"

        result = await self._sandbox.commands.run(cmd)
        if result.exit_code != 0:
            raise FileNotFoundError(f"Failed to read {path}: {result.stderr}")

        if format == "text":
            return result.stdout
        else:
            return result.stdout.encode()

    async def write(self,
                    path: str,
                    content: Union[str, bytes],
                    user: Optional[str] = None):
        """
        写入文件内容

        Args:
            path: 文件路径
            content: 文件内容
            user: 运行用户
        """
        # 使用 base64 编码避免特殊字符问题
        if isinstance(content, str):
            content = content.encode()

        encoded = base64.b64encode(content).decode()
        cmd = f"echo {encoded} | base64 -d > {path}"
        if user:
            cmd = f"sudo -u {user} {cmd}"

        result = await self._sandbox.commands.run(cmd)
        if result.exit_code != 0:
            raise IOError(f"Failed to write {path}: {result.stderr}")

    async def list_dir(self,
                       path: str,
                       depth: int = 1) -> List[FileInfo]:
        """
        列出目录内容

        Args:
            path: 目录路径
            depth: 递归深度 (1 = 仅当前层, -1 = 无限递归)

        Returns:
            FileInfo 对象列表
        """
        request = FilesystemListDirRequest(path=path, depth=depth)
        response = await self._grpc_client.ListDir(request)

        return [
            FileInfo(
                name=item.name,
                path=item.path,
                is_dir=item.is_dir,
                is_file=item.is_file,
                mode=item.mode,
                modified=item.modified.ToDatetime(),
            )
            for item in response.entries
        ]

    async def make_dir(self, path: str):
        """创建目录 (递归)"""
        request = FilesystemMakeDirRequest(path=path)
        await self._grpc_client.MakeDir(request)

    async def remove(self, path: str):
        """删除文件或目录 (递归)"""
        request = FilesystemRemoveRequest(path=path)
        await self._grpc_client.Remove(request)

    async def move(self, src: str, dst: str):
        """移动/重命名文件"""
        request = FilesystemMoveRequest(source=src, destination=dst)
        await self._grpc_client.Move(request)

    async def watch_dir(self,
                        path: str,
                        recursive: bool = True,
                        on_event: Optional[Callable[[WatchEvent], None]] = None) -> AsyncIterator[WatchEvent]:
        """
        监控目录变化 (流式)

        Args:
            path: 监控路径
            recursive: 是否递归监控子目录
            on_event: 事件回调 (可选)

        Yields:
            WatchEvent 对象
        """
        request = FilesystemWatchDirRequest(
            path=path,
            recursive=recursive
        )

        stream = self._grpc_client.WatchDir(request)

        async for event in stream:
            watch_event = WatchEvent(
                type=event.type.name.replace("EVENT_TYPE_", ""),
                path=event.path,
                name=event.name,
                is_dir=event.is_dir,
            )

            if on_event:
                on_event(watch_event)

            yield watch_event
```

**目录监控示例:**

```python
import asyncio
from e2b import AsyncSandbox

async def main():
    sandbox = await AsyncSandbox.create()

    # 初始化项目目录
    await sandbox.files.make_dir("/workspace/project")
    await sandbox.files.write("/workspace/project/main.py", "print('Hello')")

    # 监控项目目录变化
    async def on_file_change(event: WatchEvent):
        print(f"[{event.type}] {event.path}")

        if event.type == "CREATE" and event.path.endswith(".py"):
            # 自动运行新创建的 Python 文件
            result = await sandbox.commands.run(f"python {event.path}")
            print(result.stdout)

    # 启动监控 (后台任务)
    watch_task = asyncio.create_task(
        sandbox.files.watch_dir(
            "/workspace/project",
            recursive=True,
            on_event=on_file_change
        )
    )

    # 模拟文件操作
    await asyncio.sleep(1)
    await sandbox.files.write("/workspace/project/test.py", "print('Test')")
    await asyncio.sleep(1)
    await sandbox.files.remove("/workspace/project/main.py")
    await asyncio.sleep(1)

    watch_task.cancel()
    await sandbox.kill()

asyncio.run(main())
```

#### 特点三: 细粒度网络控制 - 安全与灵活的平衡

**设计目标:** AI 生成的代码可能需要访问外部 API,但必须受到严格控制

**网络配置模型** (`packages/python-sdk/e2b/sandbox/sandbox_api.py`):

```python
class SandboxNetworkConfig:
    """Sandbox 网络配置"""

    def __init__(self,
                 allow_out: Optional[List[str]] = None,
                 deny_out: Optional[List[str]] = None,
                 allow_public_traffic: bool = True,
                 mask_request_host: Optional[str] = None):
        """
        Args:
            allow_out: 允许的出站地址 (CIDR 或域名)
            deny_out: 拒绝的出站地址 (优先级高于 allow_out)
            allow_public_traffic: 是否允许公网访问 Sandbox
            mask_request_host: 请求主机掩码 (支持 ${PORT} 变量)
        """
        self.allow_out = allow_out or []
        self.deny_out = deny_out or []
        self.allow_public_traffic = allow_public_traffic
        self.mask_request_host = mask_request_host


class SandboxNetworkOpts:
    """Sandbox 创建时的网络选项"""

    @staticmethod
    def allow_all():
        """允许所有出站流量 (默认)"""
        return SandboxNetworkConfig()

    @staticmethod
    def deny_all():
        """拒绝所有出站流量"""
        return SandboxNetworkConfig(
            deny_out=["0.0.0.0/0"],
            allow_public_traffic=False
        )

    @staticmethod
    def whitelist(allowed_hosts: List[str]):
        """仅允许特定主机"""
        return SandboxNetworkConfig(
            allow_out=allowed_hosts,
            deny_out=["0.0.0.0/0"]  # 默认拒绝所有,仅允许白名单
        )

    @staticmethod
    def blacklist(denied_hosts: List[str]):
        """拒绝特定主机,允许其他"""
        return SandboxNetworkConfig(
            deny_out=denied_hosts
        )
```

```python
        "172.16.0.0/12",    # 拒绝内网 B 类
        "192.168.0.0/16",   # 拒绝内网 C 类
        "169.254.0.0/16",   # 拒绝链路本地地址
        "127.0.0.0/8"       # 拒绝 localhost
    ])
)

# 场景 4: 公网不可访问 (仅内部调用)
sandbox = await AsyncSandbox.create(
    template="base",
    network=SandboxNetworkConfig(
        allow_public_traffic=False  # 无法从公网访问 Sandbox
    )
)
```

**底层实现机制:**

E2B 通过 Firecracker VM 的网络配置实现流量控制:

1. **出站过滤**: 在 VM 网关(host)上使用 iptables 规则过滤
2. **DNS 拦截**: 对域名进行 DNS 解析后应用 CIDR 规则
3. **入站隔离**: 默认 NAT,除非 `allow_public_traffic=true`

```bash
# 示例 iptables 规则 (由 E2B 后端自动生成)

# 允许 DNS 查询
iptables -A OUTPUT -p udp --dport 53 -j ACCEPT

# 允许白名单 CIDR
iptables -A OUTPUT -d 1.1.1.1/32 -j ACCEPT
iptables -A OUTPUT -d 8.8.8.8/32 -j ACCEPT

# 拒绝黑名单 CIDR
iptables -A OUTPUT -d 10.0.0.0/8 -j DROP
iptables -A OUTPUT -d 172.16.0.0/12 -j DROP

# 默认策略
iptables -P OUTPUT DROP  # 默认拒绝所有出站 (白名单模式)
# 或
iptables -P OUTPUT ACCEPT  # 默认允许所有出站 (黑名单模式)
```

**优势:**
- **灵活控制**: 支持白名单、黑名单、混合模式
- **CIDR 支持**: 精确到 IP 段的控制
- **域名支持**: 自动解析域名为 IP
- **默认安全**: 默认拒绝内网访问,防止 SSRF

## 二、性能优化原理

### 2.1 Firecracker VM 快照机制

**冷启动 vs. 快照启动:**

| 启动方式 | 时间 | 步骤 |
|---------|------|------|
| **冷启动** | 3-5s | 1. 启动 Firecracker<br>2. 加载 Kernel<br>3. Init 进程启动<br>4. 启动 envd<br>5. 加载 Python/Node.js 运行时 |
| **快照启动** | 100-300ms | 1. 加载快照<br>2. 恢复内存状态<br>3. Resume CPU |

**快照创建流程:**

```python
# E2B 后端在模板首次使用时创建快照

# 1. 冷启动 VM
vm = firecracker.VM(
    kernel="vmlinux",
    rootfs="rootfs.ext4",
    memory_size_mb=512
)
vm.start()

# 2. 等待 envd 就绪
await vm.wait_for_agent_ready(timeout=30)

# 3. 创建快照
snapshot = vm.create_snapshot(
    snapshot_type="Full",  # 完整快照(内存 + 磁盘)
    snapshot_path="/snapshots/template-abc123.snap",
    mem_file_path="/snapshots/template-abc123.mem"
)

# 4. 后续启动直接从快照恢复
vm_new = firecracker.VM.restore_from_snapshot(
    snapshot_path="/snapshots/template-abc123.snap",
    mem_file_path="/snapshots/template-abc123.mem"
)
vm_new.resume()  # 立即可用
```

**快照优化技术:**

1. **内存去重**: 多个相同模板的 Sandbox 共享只读内存页 (CoW)
2. **磁盘 CoW**: 使用 overlayfs,只有修改的数据块占用空间
3. **增量快照**: 支持基于基础快照的增量快照(未来计划)

### 2.2 gRPC Connect 协议优化

**为什么选择 gRPC Connect?**

| 协议 | 优势 | 劣势 |
|-----|------|------|
| **REST** | 简单,广泛支持 | 无流式,JSON 序列化慢 |
| **gRPC (原生)** | 高性能,流式 | 需要 HTTP/2, 浏览器不友好 |
| **gRPC Connect** | 高性能 + 浏览器兼容 | 较新,生态小 |

**Connect 协议特点:**

1. **兼容 HTTP/1.1 和 HTTP/2**: 可在浏览器中直接使用
2. **Protocol Buffers**: 类型安全 + 高效序列化
3. **双向流**: 支持服务端流、客户端流、双向流
4. **JSON 映射**: 可选 JSON 编码,便于调试

**性能数据:**

```
基准测试: 传输 1MB 数据

REST + JSON: 45ms (序列化) + 网络延迟
gRPC Connect + Protobuf: 8ms (序列化) + 网络延迟

性能提升: 5.6x
```

## 三、安全机制

### 3.1 多层隔离

**隔离层级:**

```
┌─────────────────────────────────────────────────────────┐
│ Layer 1: 硬件虚拟化 (Firecracker + KVM)                 │
│ - 完全隔离的 Guest Kernel                               │
│ - 独立的地址空间                                        │
│ - 无法访问 Host 内核                                    │
└────────────────────────┬────────────────────────────────┘
                         │
┌────────────────────────▼────────────────────────────────┐
│ Layer 2: Seccomp 过滤                                   │
│ - 限制 Firecracker 进程的系统调用                       │
│ - 仅允许 VM 管理必需的调用                              │
└────────────────────────┬────────────────────────────────┘
                         │
┌────────────────────────▼────────────────────────────────┐
│ Layer 3: 网络隔离                                       │
│ - NAT 模式,VM 无法直接访问 Host                         │
│ - iptables 规则过滤出站流量                             │
│ - 可选:完全断网 (air-gapped)                            │
└────────────────────────┬────────────────────────────────┘
                         │
┌────────────────────────▼────────────────────────────────┐
│ Layer 4: 文件系统隔离                                   │
│ - 独立的 ext4 rootfs                                    │
│ - 无法访问 Host 文件系统                                │
│ - 可选:只读 rootfs + tmpfs overlay                      │
└────────────────────────┬────────────────────────────────┘
                         │
┌────────────────────────▼────────────────────────────────┐
│ Layer 5: 资源限制                                       │
│ - CPU 配额 (cgroups)                                    │
│ - 内存限制 (OOM killer)                                 │
│ - 磁盘配额                                              │
└─────────────────────────────────────────────────────────┘
```

### 4.1 LLM Code Interpreter

**场景**: ChatGPT Code Interpreter, Claude Code Execution

```python
from openai import OpenAI
from e2b import AsyncSandbox

client = OpenAI()
sandbox = await AsyncSandbox.create(template="base")

messages = [
    {"role": "system", "content": "You are a helpful assistant with code execution capability."},
    {"role": "user", "content": "Plot a sine wave"}
]

while True:
    # 1. LLM 生成响应
    response = client.chat.completions.create(
        model="gpt-4",
        messages=messages,
        tools=[{
            "type": "function",
            "function": {
                "name": "execute_code",
                "description": "Execute Python code in a sandbox",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "code": {"type": "string"}
                    }
                }
            }
        }]
    )

    message = response.choices[0].message

    # 2. 检查是否需要执行代码
    if message.tool_calls:
        for tool_call in message.tool_calls:
            if tool_call.function.name == "execute_code":
                code = json.loads(tool_call.function.arguments)["code"]

                # 3. 在 E2B 沙箱中执行
                result = await sandbox.commands.run(f"python -c '{code}'")

                # 4. 将结果返回给 LLM
                messages.append({
                    "role": "tool",
                    "tool_call_id": tool_call.id,
                    "content": result.stdout or result.stderr
                })
    else:
        # 5. LLM 完成任务
        print(message.content)
        break

await sandbox.kill()
```
